function useMyLocation() { const cityInput = document.getElementById('city'); const loading = document.getElementById('loading'); if (!navigator.geolocation) { alert('Geolocation not supported'); return; } loading.style.display = 'block'; // 🔥 Promise wrapper (more reliable) function getPosition() { return new Promise((resolve, reject) => { navigator.geolocation.getCurrentPosition(resolve, reject, { enableHighAccuracy: false, timeout: 15000, maximumAge: 60000 }); }); } getPosition() .then(async (position) => { const lat = position.coords.latitude; const lon = position.coords.longitude; try { // 🔹 Try OpenWeather first let res = await fetch(`https://api.openweathermap.org/data/2.5/weather?lat=${lat}&lon=${lon}&units=metric&appid=${API_KEY}`); let data = await res.json(); let locationName = ''; if (data && data.cod === 200 && data.name) { locationName = `${data.name},${data.sys.country}`; } else { // 🔹 Fallback (OpenStreetMap) const geoRes = await fetch(`https://nominatim.openstreetmap.org/reverse?format=json&lat=${lat}&lon=${lon}`); const geoData = await geoRes.json(); locationName = geoData.address.city || geoData.address.town || geoData.address.village || geoData.address.state || 'Your Location'; } // 🔥 FINAL SET (guaranteed) cityInput.value = locationName; document.getElementById('tbtLocation').value = locationName; // 🔹 Trigger main logic checkHeat(); } catch (err) { console.error(err); alert('Unable to fetch location data'); } finally { loading.style.display = 'none'; } }) .catch((error) => { loading.style.display = 'none'; if (error.code === 1) { alert('Permission denied. Allow location access.'); } else if (error.code === 2) { alert('Location unavailable.'); } else if (error.code === 3) { alert('Timeout. Try again.'); } else { alert('Error getting location.'); } }); }
